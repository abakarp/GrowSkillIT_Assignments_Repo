package dailyassigments;

public class Assigment1 {
	public static void main(String[] args) {
		/*Write a Java program to store the following details using appropriate data types and display them:
			Student ID = 101
			Grade = A
			Passed = true
			Age = 18
			Print all values on the console.*/		
		System.out.println("******Data Type Examples******");
		
		int student_id = 101;
		System.out.println("STUDENT ID = "+student_id);
		
		char grade = 'A';
		System.out.println("GRADE = "+grade);
		
		boolean result = true;
		System.out.println("RESULT PASSED = "+result);
		
		byte age = 18;
		System.out.println("AGE = "+age);				
	}
}
